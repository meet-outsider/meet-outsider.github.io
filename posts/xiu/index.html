bit_errors

自定义错误类型枚举

pub enum BitErrorValue {
    #[fail(display = "bytes read error\n")]
    BytesReadError(BytesReadError),
    #[fail(display = "bytes write error\n")]
    BytesWriteError(BytesWriteError),
    #[fail(display = "the size is bigger than 64\n")]
    TooBig,
    #[fail(display = "cannot write the whole 8 bits\n")]
    CannotWrite8Bit,
    #[fail(display = "cannot read byte\n")]
    CannotReadByte,
}

定义了结构体BitError，实现了From trait，可以将其他类型转换为BitError类型

Form trait的作用:是官方提供的一个trait，用于类型转换，可以将其他类型转换为自定义类型

#[derive(Debug)]
pub struct BitError {
    pub value: BitErrorValue,
}

impl From<BitErrorValue> for BitError {
    fn from(val: BitErrorValue) -> Self {
        BitError { value: val }
    }
}

impl From<BytesReadError> for BitError {
    fn from(error: BytesReadError) -> Self {
        BitError {
            value: BitErrorValue::BytesReadError(error),
        }
    }
}

impl From<BytesWriteError> for BitError {
    fn from(error: BytesWriteError) -> Self {
        BitError {
            value: BitErrorValue::BytesWriteError(error),
        }
    }
}

实现了Display和Fail trait,可以将BitError类型转换为字符串,并且可以获取错误的原因和回溯信息，用于调试

impl fmt::Display for BitError {
    fn fmt(&self, f: &mut fmt::Formatter) -> fmt::Result {
        fmt::Display::fmt(&self.value, f)
    }
}

impl Fail for BitError {
    fn cause(&self) -> Option<&dyn Fail> {
        self.value.cause()
    }

    fn backtrace(&self) -> Option<&Backtrace> {
        self.value.backtrace()
    }
}

bit_reader

定义了结构体BitsReader

pub struct BitsReader {
    // 读取的字节流
    reader: BytesReader,
    // 当前字节
    cur_byte: u8,
    // 当前字节剩余位数
    cur_bit_left: u8,
}

impl BitsReader {
    // 构造函数
    pub fn new(reader: BytesReader) -> Self {
        Self {
            reader,
            cur_byte: 0,
            cur_bit_left: 0,
        }
    }
    // 扩展字节流
    pub fn extend_data(&mut self, bytes: BytesMut) {
        self.reader.extend_from_slice(&bytes[..]);
    }

    // 获取字节流长度
    pub fn len(&self) -> usize {
        self.reader.len() * 8 + self.cur_bit_left as usize
    }

    // 判断字节流是否为空
    pub fn is_empty(&self) -> bool {
        self.len() == 0
    }

    // 读取字节
    pub fn read_byte(&mut self) -> Result<u8, BitError> {
        if self.cur_bit_left != 0 {
            return Err(BitError {
                value: BitErrorValue::CannotReadByte,
            });
        }

        let byte = self.reader.read_u8()?;
        Ok(byte)
    }

    // 读取位
    pub fn read_bit(&mut self) -> Result<u8, BitError> {
        if self.cur_bit_left == 0 {
            self.cur_byte = self.reader.read_u8()?;
            self.cur_bit_left = 8;
        }
        self.cur_bit_left -= 1;
        Ok((self.cur_byte >> self.cur_bit_left) & 0x01)
    }

    // 读取n位
    pub fn read_n_bits(&mut self, n: usize) -> Result<u64, BitError> {
        let mut result: u64 = 0;
        for _ in 0..n {
            result <<= 1;
            let cur_bit = self.read_bit()?;
            result |= cur_bit as u64;
        }
        Ok(result)
    }

    pub fn bits_aligment_8(&mut self) {
        self.cur_bit_left = 0;
    }
}
